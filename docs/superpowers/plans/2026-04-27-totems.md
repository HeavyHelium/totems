# totems Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Python desktop app that pops up an always-on-top "soft block" window every 45 minutes of wall-clock time, showing a quote, wisdom reminders, today's agenda, and a totem symbol — dismissible early by typing a user-set ritual phrase.

**Architecture:** One Python process; five small modules (`scheduler`, `content`, `totem_symbols`, `block_window`, `config`) plus a `duty_sources` package with a `DutySource` protocol. Modules talk through a `BlockContent` dataclass. Scheduler injects `time.sleep`/`time.monotonic` for testability. Tkinter for the window; stdlib `urllib` for the cataas fallback.

**Tech Stack:** Python 3.12, tkinter (stdlib), `urllib.request` (stdlib), `tomllib` (stdlib), pytest (dev), `uv` for env/build.

**Spec reference:** `docs/superpowers/specs/2026-04-27-totems-design.md`

```python
# src/totems/config.py
from __future__ import annotations

import os
import tomllib
from dataclasses import dataclass
from pathlib import Path


class ConfigError(Exception):
    """Raised when config.toml is missing required fields or malformed."""


@dataclass(frozen=True)
class Config:
    ritual_phrase: str
    work_minutes: int = 45
    block_minutes: int = 5
    duty_source_kind: str = "textfile"


def user_config_dir() -> Path:
    xdg = os.environ.get("XDG_CONFIG_HOME")
    base = Path(xdg) if xdg else Path(os.environ["HOME"]) / ".config"
    return base / "totems"


def load_config(path: Path) -> Config:
    try:
        data = tomllib.loads(path.read_text())
    except (OSError, tomllib.TOMLDecodeError) as e:
        raise ConfigError(f"Could not read {path}: {e}") from e

    if "ritual_phrase" not in data:
        raise ConfigError(f"{path} is missing required key 'ritual_phrase'")

    timing = data.get("timing", {})
    duty = data.get("duty_source", {})

    return Config(
        ritual_phrase=str(data["ritual_phrase"]),
        work_minutes=int(timing.get("work_minutes", 45)),
        block_minutes=int(timing.get("block_minutes", 5)),
        duty_source_kind=str(duty.get("kind", "textfile")),
    )


def write_default_config(path: Path, *, ritual_phrase: str) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(
        f'ritual_phrase = "{ritual_phrase}"\n'
        "\n"
        "[timing]\n"
        "work_minutes = 45\n"
        "block_minutes = 5\n"
        "\n"
        "[duty_source]\n"
        'kind = "textfile"\n'
    )
```

```python
# src/totems/content.py
from __future__ import annotations

import random
from dataclasses import dataclass
from importlib import resources
from pathlib import Path


LAST_RESORT_QUOTE = "Take a breath."


@dataclass(frozen=True)
class BlockContent:
    quote: str
    wisdom: list[str]
    duties: list[str]
    symbol_path: Path | None


def _read_default(filename: str) -> str:
    return resources.files("totems.defaults").joinpath(filename).read_text()


def _parse_lines(text: str) -> list[str]:
    out = []
    for line in text.splitlines():
        stripped = line.strip()
        if not stripped or stripped.startswith("#"):
            continue
        out.append(stripped)
    return out


def _merge(default_text: str, user_text: str | None) -> list[str]:
    items = _parse_lines(default_text)
    if user_text:
        items.extend(_parse_lines(user_text))
    seen: set[str] = set()
    out: list[str] = []
    for item in items:
        if item not in seen:
            seen.add(item)
            out.append(item)
    return out


def load_quotes(user_text: str | None) -> list[str]:
    return _merge(_read_default("quotes.txt"), user_text)


def load_wisdom(user_text: str | None) -> list[str]:
    return _merge(_read_default("wisdom.txt"), user_text)


def pick_quote(quotes: list[str], rng: random.Random) -> str:
    if not quotes:
        return LAST_RESORT_QUOTE
    return rng.choice(quotes)


def pick_wisdom(wisdom: list[str], rng: random.Random, n: int) -> list[str]:
    if not wisdom:
        return []
    n = min(n, len(wisdom))
    return rng.sample(wisdom, n)
```

```python
from __future__ import annotations

from pathlib import Path


class TextFileDutySource:
    def __init__(self, path: Path) -> None:
        self._path = path

    def today(self) -> list[str]:
        try:
            text = self._path.read_text()
        except (OSError, IsADirectoryError):
            return []
        out = []
        for line in text.splitlines():
            stripped = line.strip()
            if not stripped or stripped.startswith("#"):
                continue
            out.append(stripped)
        return out
```

`totem_symbols.py` exposes `get_totem_symbol(config_dir, rng, urlopen=urllib.request.urlopen) -> Path | None`:
- If `<config_dir>/totem_symbols/` has any files matching `.jpg .jpeg .png .gif .webp` (case-insensitive, excluding the `.cache/` dir), pick one with `rng`.
- Else, try `urlopen("https://cataas.com/cat")`, write bytes to `<config_dir>/totem_symbols/.cache/symbol-<n>.jpg`, return that path.
- On any HTTP error, return `None`.

The `urlopen` parameter is the test seam — no real network in tests.

```python
# src/totems/totem_symbols.py
from __future__ import annotations

import random
import time
import urllib.request
from collections.abc import Callable
from pathlib import Path
from typing import IO


IMAGE_EXTS = {".jpg", ".jpeg", ".png", ".gif", ".webp"}
FALLBACK_URL = "https://cataas.com/cat"


UrlOpen = Callable[..., IO[bytes]]


def _list_local_symbols(cats_dir: Path) -> list[Path]:
    if not cats_dir.is_dir():
        return []
    out: list[Path] = []
    for child in cats_dir.iterdir():
        if child.is_dir():
            continue  # skip .cache and any other dirs
        if child.suffix.lower() in IMAGE_EXTS:
            out.append(child)
    return sorted(out)


def _next_cache_path(cache_dir: Path) -> Path:
    cache_dir.mkdir(parents=True, exist_ok=True)
    # Use a monotonic timestamp so deletions don't cause filename collisions.
    return cache_dir / f"symbol-{time.time_ns()}.jpg"


def get_totem_symbol(
    *,
    config_dir: Path,
    rng: random.Random,
    urlopen: UrlOpen = urllib.request.urlopen,
) -> Path | None:
    symbols_dir = config_dir / "totem_symbols"
    locals_ = _list_local_symbols(cats_dir)
    if locals_:
        return rng.choice(locals_)

    try:
        with urlopen(FALLBACK_URL, timeout=5) as resp:
            data = resp.read()
    except Exception:
        return None

    out = _next_cache_path(cats_dir / ".cache")
    out.write_bytes(data)
    return out
```

The scheduler:
- Takes work/block durations (in seconds, not minutes — easier to test) and two callbacks:

```python
# src/totems/scheduler.py
from __future__ import annotations

import time
from collections.abc import Callable


BlockResult = str  # "phrase" | "timeout"


class Scheduler:
    def __init__(
        self,
        *,
        work_seconds: float,
        on_block: Callable[[], BlockResult],
        sleep: Callable[[float], None] = time.sleep,
        now: Callable[[], float] = time.monotonic,
    ) -> None:
        self._work_seconds = work_seconds
        self._on_block = on_block
        self._sleep = sleep
        self._now = now
        self._stopped = False

    def stop(self) -> None:
        self._stopped = True

    def run_once(self) -> BlockResult:
        self._sleep(self._work_seconds)
        return self._on_block()

    def run(self) -> None:
        while not self._stopped:
            self.run_once()
```

```python
# src/totems/block_window.py
from __future__ import annotations

import tkinter as tk
from tkinter import ttk

from .content import BlockContent


WIDTH = 700
HEIGHT = 450


class BlockWindow:
    def __init__(
        self,
        *,
        content: BlockContent,
        ritual_phrase: str,
        block_seconds: int,
    ) -> None:
        self._content = content
        self._phrase = ritual_phrase
        self._remaining = block_seconds
        self.reason: str | None = None

        self.root = tk.Tk()
        self.root.title("totems")
        self.root.attributes("-topmost", True)
        self.root.resizable(False, False)
        self.root.protocol("WM_DELETE_WINDOW", lambda: None)
        self._center(WIDTH, HEIGHT)

        self._build_ui()
        self.root.after(1000, self._tick)

    def _center(self, w: int, h: int) -> None:
        self.root.update_idletasks()
        sw = self.root.winfo_screenwidth()
        sh = self.root.winfo_screenheight()
        x = (sw - w) // 2
        y = (sh - h) // 2
        self.root.geometry(f"{w}x{h}+{x}+{y}")

    def _build_ui(self) -> None:
        outer = ttk.Frame(self.root, padding=12)
        outer.pack(fill="both", expand=True)

        # Top: timer
        self._timer_var = tk.StringVar(value=self._format_remaining())
        ttk.Label(outer, textvariable=self._timer_var, foreground="#888").pack(anchor="ne")

        # Middle: side-by-side
        body = ttk.Frame(outer)
        body.pack(fill="both", expand=True, pady=(8, 8))

        if self._content.symbol_path is not None:
            try:
                self._cat_img = tk.PhotoImage(file=str(self._content.symbol_path))
                cat_label = ttk.Label(body, image=self._cat_img)
                cat_label.pack(side="left", padx=(0, 12))
            except tk.TclError:
                pass  # bad image — silently skip the totem symbol panel

        right = ttk.Frame(body)
        right.pack(side="left", fill="both", expand=True)

        ttk.Label(
            right,
            text=self._content.quote,
            wraplength=400,
            font=("TkDefaultFont", 12, "italic"),
        ).pack(anchor="w", pady=(0, 12))

        if self._content.wisdom:
            ttk.Label(right, text="Wisdom", font=("TkDefaultFont", 9, "bold")).pack(anchor="w")
            for w in self._content.wisdom:
                ttk.Label(right, text=f"· {w}").pack(anchor="w")

        if self._content.duties:
            ttk.Label(right, text="Today", font=("TkDefaultFont", 9, "bold")).pack(
                anchor="w", pady=(8, 0)
            )
            for d in self._content.duties:
                ttk.Label(right, text=f"· {d}").pack(anchor="w")

        # Bottom: phrase entry
        self._entry = ttk.Entry(outer)
        self._entry.pack(fill="x", pady=(8, 0))
        self._entry.bind("<Return>", lambda _e: self._on_submit())

    def _format_remaining(self) -> str:
        m, s = divmod(max(0, self._remaining), 60)
        return f"{m:02d}:{s:02d} remaining"

    def _tick(self) -> None:
        self._remaining -= 1
        self._timer_var.set(self._format_remaining())
        if self._remaining <= 0:
            self._close("timeout")
            return
        self.root.after(1000, self._tick)

    def _on_submit(self) -> None:
        if self._entry.get() == self._phrase:
            self._close("phrase")

    def _close(self, reason: str) -> None:
        if self.reason is None:
            self.reason = reason
            self.root.quit()

    def run(self) -> str:
        self.root.mainloop()
        self.root.destroy()
        return self.reason or "timeout"
```

```python
from __future__ import annotations

import argparse
import random
import sys
from pathlib import Path

from .block_window import BlockWindow
from .totem_symbols import get_totem_symbol
from .config import Config, ConfigError, load_config, user_config_dir, write_default_config
from .content import BlockContent, load_quotes, load_wisdom, pick_quote, pick_wisdom
from .duty_sources import make_duty_source
from .scheduler import Scheduler


def _read_text_or_none(p: Path) -> str | None:
    try:
        return p.read_text()
    except OSError:
        return None


def _build_block_content(cfg_dir: Path, cfg: Config, rng: random.Random) -> BlockContent:
    quotes = load_quotes(_read_text_or_none(cfg_dir / "quotes.txt"))
    wisdom_pool = load_wisdom(_read_text_or_none(cfg_dir / "wisdom.txt"))
    duty_source = make_duty_source(cfg.duty_source_kind, config_dir=cfg_dir)
    return BlockContent(
        quote=pick_quote(quotes, rng),
        wisdom=pick_wisdom(wisdom_pool, rng, n=2),
        duties=duty_source.today(),
        symbol_path=get_totem_symbol(config_dir=cfg_dir, rng=rng),
    )


def _first_run(config_path: Path) -> None:
    print("First-run setup for totems.")
    print("Pick a ritual phrase you'll type to dismiss a block early.")
    print("Long sentences work best — the friction is the point.")
    while True:
        phrase = input("Ritual phrase: ").strip()
        if phrase:
            break
        print("Phrase can't be empty.")
    write_default_config(config_path, ritual_phrase=phrase)
    print(f"Wrote {config_path}. You can edit it later.")


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(prog="totems")
    parser.add_argument("--debug-now", action="store_true", help="trigger one block immediately and exit")
    parser.add_argument("--fast", action="store_true", help="use 5-second timers for end-to-end smoke testing")
    args = parser.parse_args(argv)

    cfg_dir = user_config_dir()
    config_path = cfg_dir / "config.toml"
    if not config_path.exists():
        _first_run(config_path)

    try:
        cfg = load_config(config_path)
    except ConfigError as e:
        print(f"totems: config error: {e}", file=sys.stderr)
        return 2

    rng = random.Random()

    def trigger_block() -> str:
        content = _build_block_content(cfg_dir, cfg, rng)
        block_seconds = 5 if args.fast else cfg.block_minutes * 60
        win = BlockWindow(content=content, ritual_phrase=cfg.ritual_phrase, block_seconds=block_seconds)
        return win.run()

    if args.debug_now:
        trigger_block()
        return 0

    work_seconds = 5 if args.fast else cfg.work_minutes * 60
    sched = Scheduler(work_seconds=work_seconds, on_block=trigger_block)
    try:
        sched.run()
    except KeyboardInterrupt:
        print("\ntotems: bye")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
